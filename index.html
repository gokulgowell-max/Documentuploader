Google Sheets Backend Connection

Link your Gowell Spreadsheet as the live backend database. Changes will be loaded directly from your sheet.

Get ID from URL: docs.google.com/spreadsheets/d/[SPREADSHEET_ID]/edit
Provide Web App URL to write back edits. If empty, the sheet is Read-Only.
How to set up Google Sheets Read/Write Web App (Instructions & Apps Script Code)

1. Open your Google Spreadsheet.

2. Go to Extensions > Apps Script.

3. Delete any default code, paste the script code below, and click Save (💾 icon):

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var teamsSheet = ss.getSheetByName("Teams") || ss.insertSheet("Teams");
  var projectsSheet = ss.getSheetByName("Projects") || ss.insertSheet("Projects");
  
  var teamsData = getSheetRowsAsObjects(teamsSheet);
  var projectsData = getSheetRowsAsObjects(projectsSheet);
  
  var result = {
    teams: teamsData,
    projects: projectsData
  };
  
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var params = JSON.parse(e.postData.contents);
  
  if (params.action === "pushAll") {
    writeObjectsToSheet(ss, "Teams", params.teams, ["id", "name", "icon", "description"]);
    writeObjectsToSheet(ss, "Projects", params.projects, ["id", "teamId", "title", "status", "summary", "details", "updatedAt"]);
    return ContentService.createTextOutput(JSON.stringify({success: true, message: "Data synchronized in spreadsheet successfully!"}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  return ContentService.createTextOutput(JSON.stringify({success: false, message: "Unknown action"}))
    .setMimeType(ContentService.MimeType.JSON);
}

function getSheetRowsAsObjects(sheet) {
  var data = sheet.getDataRange().getValues();
  if (data.length <= 1) return [];
  var headers = data[0];
  var result = [];
  for (var i = 1; i < data.length; i++) {
    var obj = {};
    for (var j = 0; j < headers.length; j++) {
      obj[headers[j]] = data[i][j];
    }
    result.push(obj);
  }
  return result;
}

function writeObjectsToSheet(ss, sheetName, objects, headers) {
  var sheet = ss.getSheetByName(sheetName);
  if (sheet) {
    sheet.clear();
  } else {
    sheet = ss.insertSheet(sheetName);
  }
  sheet.appendRow(headers);
  if (!objects || objects.length === 0) return;
  var rows = [];
  for (var i = 0; i < objects.length; i++) {
    var obj = objects[i];
    var row = [];
    for (var j = 0; j < headers.length; j++) {
      row.push(obj[headers[j]] || "");
    }
    rows.push(row);
  }
  sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
}

4. Click Deploy > New deployment in Apps Script.

5. Click the Cog gear settings icon, select Web app.

6. Change: Execute as to "Me", and Who has access to "Anyone".

7. Click Deploy, authorize the permissions, and copy the **Web App URL**.

8. Paste that copied Web App URL in the field above and save settings!
